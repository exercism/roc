import pf.Stdout

factorial = \n ->
    if n == 1 then
        1
    else
        n * factorial (n - 1)

main =
    result = factorial 5
    Stdout.line! "factorial 5 = 1 * 2 * 3 * 4 * 5 = $(Num.toStr result)"
