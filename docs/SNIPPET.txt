import pf.Stdout

factorial = \number ->
    when number is
        1 -> 1
        n -> n * factorial (n - 1)

expect factorial 5 == 1 * 2 * 3 * 4 * 5

main! = \_args ->
    result = factorial 20 |> Num.toStr
    Stdout.line! "factorial 20 = $(result)"
